def sum_mul(choice, *val):
    if choice =="sum":
        result =0
        for i in val:
            result =result +i
    elif choice == "mul":
        result = 1
        for i in val:
            result = result * i
    return result

print(sum_mul("sum",1,2,3))
print(sum_mul("mul",1,2,3,4,5))
