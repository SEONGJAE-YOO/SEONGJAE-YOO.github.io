word_freq = {}

  if word.text.lower() not in  stop_words:
    if word.text.lower() not in punctuation:
      if word.text not in word_freq.keys(): # 단어 몇번 나왔는지 횟수 
        word_freq[word.text] = 1
      else:
        word_freq[word.text] += 1
